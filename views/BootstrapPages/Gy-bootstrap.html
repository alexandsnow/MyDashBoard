there is a problem when I use the tooltip.js with angular directly. you can solve this by visit the website stackoverflow 
           <button type="button"
                class="btn btn-default"
                data-toggle="tooltip"
                data-placement="left"
                title="Tooltip on left">
                Tooltip on left
            </button>
        
           <a href="#"
                class="tooltip-test"
                data-toggle="tooltip"
                title="default Tooltip"
                tooltip>
                default Tooltip.
            </a>
        
            $(function () {
                $("[data-toggle='tooltip']").tooltip();
                    });
if I use the code above,there will no a tooltip show up! it is because the conflict between angular and bootstrap !  and I solve the problem with two ways.

easy solution

            //add onmouseenter attribution on the element
            onmouseenter="$(this).tooltip('show')"

            //use example
            <button type="button"
                onmouseenter="$(this).tooltip('show')"
                class="btn btn-default"
                data-toggle="tooltip"
                data-placement="right"
                title="Tooltip on right">
            Tooltip on right
            </button>

            //effect

custom directive

            // you can custom your directive
            myApp.directive('tooltip', function(){
                return {
                    restrict: 'A',
                    link: function(scope, element, attrs){
                        $(element).hover(function(){
                            // on mouseenter
                            $(element).tooltip('show');
                        }, function(){
                            // on mouseleave
                            $(element).tooltip('hide');
                        });
                    }
                };
                });
            // you can use this like this
            <a href="#"
                class="tooltip-test"
                data-toggle="tooltip"
                title="default Tooltip" tooltip>
                default Tooltip
            </a>
